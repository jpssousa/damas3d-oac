void game_loop ()
{
	int player_turn = 0;

	while (1)
	{
		play(player_turn++);
		if (!check_life()) game_over();

		player_turn %= 2;
	}
}

void play (player_turn)
{
	 bool need_to_eat = preprocessing(player_turn);
	
	do
	{
		x = input();
		y = input();
	} while (validate_move(x, y) || validate_capture(x, y));

	postprocessing(x, y, need_to_eat, player_turn, address_of_eaten_piece);
}
